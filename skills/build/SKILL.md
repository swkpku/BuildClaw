```python
#!/usr/bin/env python3
"""
Personal Assistant Bot
Generated by Claude Code via BuildClaw — you own every line.
Audit time: ~20 min | Dependencies: listed in requirements.txt
"""

import os
import json
from pathlib import Path
from dotenv import load_dotenv
import anthropic
from telegram import Update
from telegram.ext import Application, CommandHandler, MessageHandler, filters, ContextTypes

load_dotenv()

# ── Configuration ─────────────────────────────────────────────────────────────
TELEGRAM_TOKEN    = os.environ["TELEGRAM_TOKEN"]
ANTHROPIC_API_KEY = os.environ["ANTHROPIC_API_KEY"]
_raw_ids = os.environ["ALLOWED_USER_IDS"].split(",")
for _id in _raw_ids:
    if not _id.strip().isdigit():
        raise SystemExit(
            f"ALLOWED_USER_IDS must be numeric Telegram user IDs, not usernames.\n"
            f"  Got: {_id.strip()!r}\n"
            f"  Fix: message @userinfobot on Telegram to get your numeric ID."
        )
ALLOWED_USER_IDS  = {int(x.strip()) for x in _raw_ids}
WORKSPACE         = Path(os.environ.get("WORKSPACE", str(Path.home() / "assistant-workspace")))
MODEL             = os.environ.get("MODEL", "claude-sonnet-4-6")

# ── Security ──────────────────────────────────────────────────────────────────
#
# BLOCKED_PATTERNS is checked on every path before any filesystem operation.
# If any component of the path matches, the operation is rejected — always.
#
BLOCKED_PATTERNS = {
    ".ssh", ".gnupg", ".aws", ".azure", ".gcloud", ".kube", ".docker",
    ".env", "credentials", ".netrc", ".npmrc",
    "id_rsa", "id_ed25519", "id_ecdsa", "private_key", ".secret",
}

def is_safe_path(path: str) -> bool:
    """True only if path resolves inside WORKSPACE and no component matches BLOCKED_PATTERNS."""
    try:
        target = Path(path).resolve()
        if not str(target).startswith(str(WORKSPACE.resolve())):
            return False
        for part in target.parts:
            if any(blocked in part.lower() for blocked in BLOCKED_PATTERNS):
                return False
        return True
    except Exception:
        return False
```

```python
# ── Anthropic agent loop ───────────────────────────────────────────────────────
client = anthropic.Anthropic(api_key=ANTHROPIC_API_KEY)
conversations: dict[int, list] = {}

SYSTEM_PROMPT = f"""You are a personal AI assistant accessible via Telegram.
Your workspace is: {WORKSPACE}
[Describe selected tool capabilities here based on what was chosen]
Keep responses concise and under 4000 characters."""

def run_agent(user_id: int, user_message: str) -> str:
    history = conversations.setdefault(user_id, [])
    history.append({"role": "user", "content": user_message})
    if len(history) > 20:
        conversations[user_id] = history[-20:]
        history = conversations[user_id]
    while True:
        response = client.messages.create(
            model=MODEL,
            max_tokens=4096,
            system=SYSTEM_PROMPT,
            tools=TOOL_DEFINITIONS,
            messages=history,
        )
        history.append({"role": "assistant", "content": response.content})
        if response.stop_reason == "end_turn":
            return "\n".join(b.text for b in response.content if hasattr(b, "text")) or "(no response)"
        if response.stop_reason == "tool_use":
            results = []
            for block in response.content:
                if block.type == "tool_use":
                    fn = TOOL_DISPATCH.get(block.name)
                    results.append({
                        "type": "tool_result",
                        "tool_use_id": block.id,
                        "content": str(fn(block.input) if fn else f"Unknown tool: {block.name}"),
                    })
            history.append({"role": "user", "content": results})
        else:
            return f"(unexpected stop_reason: {response.stop_reason})"

# ── Telegram handlers ──────────────────────────────────────────────────────────
async def handle_message(update: Update, context: ContextTypes.DEFAULT_TYPE) -> None:
    if update.effective_user.id not in ALLOWED_USER_IDS:
        return  # silently ignore — no error, no log
    text = (update.message.text or "").strip()
    if not text:
        return
    thinking = await update.message.reply_text("…")
    try:
        reply = run_agent(update.effective_user.id, text)
    except Exception as exc:
        reply = f"Error: {exc}"
    finally:
        await thinking.delete()
    for i in range(0, max(len(reply), 1), 4000):
        await update.message.reply_text(reply[i:i + 4000])

async def handle_reset(update: Update, context: ContextTypes.DEFAULT_TYPE) -> None:
    if update.effective_user.id not in ALLOWED_USER_IDS:
        return
    conversations.pop(update.effective_user.id, None)
    await update.message.reply_text("Conversation cleared.")

# ── Entry point ────────────────────────────────────────────────────────────────
def main() -> None:
    WORKSPACE.mkdir(parents=True, exist_ok=True)
    print(f"Workspace : {WORKSPACE}")
    print(f"Authorized: {ALLOWED_USER_IDS}")
    print(f"Model     : {MODEL}")
    app = Application.builder().token(TELEGRAM_TOKEN).build()
    app.add_handler(CommandHandler("reset", handle_reset))
    app.add_handler(MessageHandler(filters.TEXT & ~filters.COMMAND, handle_message))
    print("Bot running. Press Ctrl+C to stop.")
    app.run_polling(drop_pending_updates=True)

if __name__ == "__main__":
    main()
```

Place `TOOL_DEFINITIONS` and `TOOL_DISPATCH` before the agent loop, containing
only the tools for selected blocks.

## Block reference — code for each block

### `files` block

**Risk:** Medium. Claude can read and write any file inside WORKSPACE.
Content of files is sent to Anthropic's API. Blocked patterns still apply.

**Imports:** none new (pathlib already present)

**Tool functions:**
```python
def tool_read_file(path: str) -> str:
    if not is_safe_path(path):
        return f"Blocked: '{path}' is outside workspace or matches a blocked pattern."
    p = Path(path)
    if not p.exists():
        return f"Error: file not found: {path}"
    return p.read_text(encoding="utf-8", errors="replace")

def tool_write_file(path: str, content: str) -> str:
    if not is_safe_path(path):
        return f"Blocked: '{path}' is outside workspace or matches a blocked pattern."
    p = Path(path)
    p.parent.mkdir(parents=True, exist_ok=True)
    p.write_text(content, encoding="utf-8")
    return f"Written {len(content)} bytes to {p}"

def tool_list_dir(path: str = "") -> str:
    target = (WORKSPACE / path).resolve() if path else WORKSPACE.resolve()
    if not str(target).startswith(str(WORKSPACE.resolve())):
        return "Blocked: path is outside workspace."
    if not target.exists():
        return f"Error: directory not found: {target}"
    items = sorted(target.iterdir(), key=lambda p: (p.is_file(), p.name))
    return "\n".join(f"{'[dir]  ' if p.is_dir() else '[file] '}{p.name}" for p in items) or "(empty)"
```

### `shell` block

**Risk:** High. Claude can run any shell command inside WORKSPACE. Commands
can make outbound network requests and consume system resources. The workspace
boundary is enforced at the subprocess level (`cwd=WORKSPACE`), not inside the
shell. Consider running the bot in Docker after adding this block.

**Imports:** `import subprocess`

**Tool function:**
```python
def tool_shell(command: str) -> str:
    """Runs inside WORKSPACE only. 30-second timeout. Output capped at 4000 chars."""
    WORKSPACE.mkdir(parents=True, exist_ok=True)
    result = subprocess.run(
        command, shell=True, capture_output=True, text=True,
        cwd=str(WORKSPACE), timeout=30,
    )
    output = (result.stdout + result.stderr).strip()
    return output[:4000] if output else "(no output)"
```

**TOOL_DEFINITIONS entry:**
```python
{"name": "shell", "description": "Run a shell command inside the workspace. Max 30 seconds.",
 "input_schema": {"type": "object", "properties": {"command": {"type": "string"}}, "required": ["command"]}},
```

**TOOL_DISPATCH entry:**
```python
"shell": lambda inp: tool_shell(inp["command"]),
```

**Tool functions:**
```python
def _load_memory() -> dict:
    try:
        return json.loads(MEMORY_FILE.read_text()) if MEMORY_FILE.exists() else {}
    except Exception:
        return {}

def tool_remember(key: str, value: str) -> str:
    m = _load_memory()
    m[key] = value
    MEMORY_FILE.write_text(json.dumps(m, indent=2))
    return f"Remembered: {key}"

def tool_recall(key: str = "") -> str:
    m = _load_memory()
    if key:
        return m.get(key, f"Nothing stored for '{key}'")
    return "\n".join(f"{k}: {v}" for k, v in m.items()) if m else "Nothing remembered yet."

def tool_forget(key: str) -> str:
    m = _load_memory()
    if key in m:
        del m[key]
        MEMORY_FILE.write_text(json.dumps(m, indent=2))
        return f"Forgotten: {key}"
    return f"Nothing to forget for '{key}'"
```

```python
def tool_web_search(query: str, max_results: int = 5) -> str:
    from duckduckgo_search import DDGS
    with DDGS() as ddgs:
        results = list(ddgs.text(query, max_results=max_results))
    if not results:

    return "\n\n".join(f"**{r['title']}**\n{r['href']}\n{r['body']}" for r in results)
```

### `scheduler` block

**Risk:** Medium. Runs tasks autonomously in the background on a schedule.
Tasks call Claude and send results to the authorized Telegram user. Requires
APScheduler and an asyncio-compatible setup.

**Imports:** `from apscheduler.schedulers.asyncio import AsyncIOScheduler`

**Constants and setup:**
```python
SCHEDULE_FILE = WORKSPACE / "schedule.json"
_scheduler = AsyncIOScheduler()
_bot_app = None  # set in main() after Application is created
_default_user_id: int = next(iter(ALLOWED_USER_IDS))  # scheduled tasks go to first authorized user
```

**Tool functions:**
```python
def _load_schedules() -> dict:
    try:
        return json.loads(SCHEDULE_FILE.read_text()) if SCHEDULE_FILE.exists() else {}
    except Exception:
        return {}

def _save_schedules(schedules: dict) -> None:
    SCHEDULE_FILE.write_text(json.dumps(schedules, indent=2))

def tool_schedule(label: str, cron: str, task: str) -> str:
    """cron format: 'minute hour day month day_of_week'  e.g. '0 8 * * *' = daily 8am"""
    schedules = _load_schedules()
    schedules[label] = {"cron": cron, "task": task}
    _save_schedules(schedules)
    _register_job(label, cron, task)
    return f"Scheduled '{label}': {cron} — {task}"

def tool_list_schedules() -> str:
    schedules = _load_schedules()
    if not schedules:
        return "No scheduled tasks."
    return "\n".join(f"{k}: [{v['cron']}] {v['task']}" for k, v in schedules.items())

def tool_unschedule(label: str) -> str:
    schedules = _load_schedules()
    if label not in schedules:
        return f"No scheduled task named '{label}'"
    del schedules[label]
    _save_schedules(schedules)
    try:
        _scheduler.remove_job(label)
    except Exception:
        pass
    return f"Removed scheduled task '{label}'"

def _register_job(label: str, cron: str, task: str) -> None:
    parts = cron.split()
    if len(parts) != 5:
        return
    minute, hour, day, month, day_of_week = parts
    try:
        _scheduler.remove_job(label)
    except Exception:
        pass
    _scheduler.add_job(
        _run_scheduled_task, "cron",
        id=label, args=[task],
        minute=minute, hour=hour, day=day, month=month, day_of_week=day_of_week,
    )

async def _run_scheduled_task(task: str) -> None:
    if _bot_app is None:
        return
    reply = run_agent(_default_user_id, f"[Scheduled task] {task}")
    await _bot_app.bot.send_message(chat_id=_default_user_id, text=reply[:4000])
```
