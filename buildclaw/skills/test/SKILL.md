---
name: test
description: Generate and run a security test suite for a personal assistant bot — use when the user wants to verify that path safety, blocked patterns, and tool guards are working correctly
---

You are going to generate `test_security.py` — a pytest test suite that proves
the security invariants in `bot.py` hold.

These tests cover the only code that actually matters for security:
`is_safe_path()`, `BLOCKED_PATTERNS`, and every tool's path enforcement.
Everything else is covered by the libraries. These are the lines that were
written by hand and can have bugs.

---

## Step 1 — Read bot.py

Read `bot.py`. Identify:

1. Which tool functions are present (`tool_read_file`, `tool_write_file`,
   `tool_list_dir`, `tool_shell`)
2. The exact name of the safe-path function (usually `is_safe_path`)
3. The exact name of the blocked patterns set (usually `BLOCKED_PATTERNS`)

If `bot.py` does not exist, stop and say:
"No bot.py found. Run /build first."

---

### Class: TestBlockedPatternsCompleteness

```python
class TestBlockedPatternsCompleteness:
    """BLOCKED_PATTERNS must contain every required entry. Missing entries are vulnerabilities."""

    REQUIRED_PATTERNS = {
        ".ssh", ".gnupg", ".aws", ".azure", ".gcloud", ".kube", ".docker",
        ".env", "credentials", ".netrc", ".npmrc",
        "id_rsa", "id_ed25519", "private_key", ".secret",
    }

    def test_all_required_patterns_present(self):
        missing = self.REQUIRED_PATTERNS - bot.BLOCKED_PATTERNS
        assert not missing, (
            f"BLOCKED_PATTERNS is missing required entries: {missing}\n"
            f"These patterns protect against credential theft. Add them."
        )

    def test_blocked_patterns_is_a_set(self):
        assert isinstance(bot.BLOCKED_PATTERNS, (set, frozenset)), (
            "BLOCKED_PATTERNS should be a set for O(1) lookup, not a list"
        )
```

class TestToolReadFile:
    """tool_read_file must reject unsafe paths before touching the filesystem."""

    def test_blocks_absolute_path_outside_workspace(self, tmp_path):
        with patch.object(bot, "WORKSPACE", tmp_path):
            result = bot.tool_read_file("/etc/passwd")
        assert "Blocked" in result

    def test_blocks_path_traversal(self, tmp_path):
        with patch.object(bot, "WORKSPACE", tmp_path):
            result = bot.tool_read_file(str(tmp_path / ".." / "secret"))
        assert "Blocked" in result

    def test_blocks_ssh_key(self, tmp_path):
        with patch.object(bot, "WORKSPACE", tmp_path):
            result = bot.tool_read_file(str(tmp_path / ".ssh" / "id_rsa"))
        assert "Blocked" in result

    def test_blocks_aws_credentials(self, tmp_path):
        with patch.object(bot, "WORKSPACE", tmp_path):
            result = bot.tool_read_file(str(tmp_path / ".aws" / "credentials"))
        assert "Blocked" in result

    def test_reads_valid_file(self, tmp_path):
        target = tmp_path / "hello.txt"
        target.write_text("hello world")
        with patch.object(bot, "WORKSPACE", tmp_path):
            result = bot.tool_read_file(str(target))
        assert result == "hello world"

    def test_returns_error_for_missing_file(self, tmp_path):
        with patch.object(bot, "WORKSPACE", tmp_path):
            result = bot.tool_read_file(str(tmp_path / "nonexistent.txt"))
        assert "Error" in result or "not found" in result.lower()

class TestToolListDir:
    """tool_list_dir must not expose directories outside the workspace."""

    def test_blocks_path_outside_workspace(self, tmp_path):
        with patch.object(bot, "WORKSPACE", tmp_path):
            result = bot.tool_list_dir("/etc")
        assert "Blocked" in result or "Error" in result

    def test_blocks_path_traversal(self, tmp_path):
        with patch.object(bot, "WORKSPACE", tmp_path):
            result = bot.tool_list_dir("../")
        assert "Blocked" in result or "Error" in result

    def test_lists_workspace_root(self, tmp_path):
        (tmp_path / "notes.txt").write_text("hi")
        (tmp_path / "subdir").mkdir()
        with patch.object(bot, "WORKSPACE", tmp_path):
            result = bot.tool_list_dir()
        assert "notes.txt" in result
        assert "subdir" in result

    def test_lists_subdirectory(self, tmp_path):
        subdir = tmp_path / "projects"
        subdir.mkdir()
        (subdir / "readme.md").write_text("hi")
        with patch.object(bot, "WORKSPACE", tmp_path):
            result = bot.tool_list_dir("projects")
        assert "readme.md" in result
